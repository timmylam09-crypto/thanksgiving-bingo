function makeTable(board){

  const table=document.createElement('table');

  let idx=0;

  for(let r=0;r<5;r++){

    const tr=document.createElement('tr');

    for(let c=0;c<3;c++){

      const td=document.createElement('td');

      td.textContent=board[idx++];

      td.addEventListener('click',()=>td.classList.toggle('marked'));

      tr.appendChild(td);

    }

    table.appendChild(tr);

  }

  return table;

}

 

function newBoard(){

  const i=Math.floor(Math.random()*boards.length);

  const board=boards[i];

  const area=document.getElementById('boardArea');

  area.innerHTML='';

  area.appendChild(makeTable(board));

  sessionStorage.setItem('bingo_index',i);

}
